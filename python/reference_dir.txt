CampusMate/

├── data/                     # 【数据层】所有的 Mock 数据在这里

│
├── tools/                    # 【工具层】Agent 的手和脚
│   ├── __init__.py
│   ├── campus_tools.py       # 定义查课表、查空教室的函数
│   ├── market_tools.py       # 定义二手搜索、上架的函数
│   └── notification_tools.py # 定义发邮件、生成日历的函数
│
├── agents/                   # 【智能层】Agent 的人设定义
│   ├── __init__.py

│   └── definitions.py        # 定义 AssistantAgent, UserProxy, GroupChat
│
├── app.py                    # 【用户前台】Chainlit 启动入口 (主程序)
├── admin.py                  # 【管理后台】Streamlit 启动入口 (上帝模式)
│
├── requirements.txt          # 依赖包列表
├── Dockerfile                # 部署配置 (给华为云用)
└── docker-compose.yml        # 一键启动脚本
